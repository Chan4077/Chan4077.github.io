For example:

# An integer
S = 11
# A list of integers
V = [1,2,5]

"""
Write a function to determine the minimum number of elements in list V that can be summed up to S
"""
def minElement(S, V, E=0):

  if S == 0: return 0
  if S < 1: return E

print(minElement(S, V))

(Note that this is Python)
